217 std::pair<std::vector<std::string>, std::vector<unsigned int>> read_imdb_sentiments(
218  std::string path ) {
219  //create container housing 12500 strings

221  std::vector<unsigned int> ratings( 25000 );
222 
223  //starts in the pos sub dir
224  std::string currentPath = path + "pos/";
225 
226  //local data structure to hold the id and rating of a specific file
227  std::pair<int, int> idScore;
228  std::string line;
229 
230  //used to hold both sentiments and ratings
231  std::pair<std::vector<std::string>, std::vector<unsigned int>> data;
232 
233  //loop over the neg and pos sub-directories
234  for ( size_t i = 0; i < 2; i++ ) {
235  try {
236  //find offset for inserting into vectors
237  int offset = i * 12500;
238 
239  //iterate over the directory in the current path and go through each file within it
240  for ( auto j = boost::filesystem::directory_iterator( currentPath );
241  j != boost::filesystem::directory_iterator(); j++ ) {
242  if ( !is_directory( j->path() ) ) //we eliminate directories
243  {
244  //grab complete file path and name
245  std::string filename(
246  currentPath + j->path().filename().string() );

252  getline( file, line );
253  line = line.substr( 0, MAX_LEN );
254 
255  //grab id and rating

257  j->path().filename().string() );

272  }
