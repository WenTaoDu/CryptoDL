25 inline uint32_t read_header(const std::unique_ptr<char[]>& buffer, size_t position) {
26  auto header = reinterpret_cast<uint32_t*>(buffer.get());
27 
28  auto value = *(header + position);
29  return (value << 24) | ((value << 8) & 0x00FF0000) | ((value >> 8) & 0X0000FF00) | (value >> 24);
30 }
