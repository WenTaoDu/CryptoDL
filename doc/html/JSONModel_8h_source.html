115  Model<ValueType, WeightType, DataTensorType, WeightTensorType> fromFile( std::string filename, MemoryUsage usage = MemoryUsage::greedy ) {
116  //read json file into json object
117  json modelConfig;
118  try {
119  std::ifstream inputFile( filename );

124  }
125 
126  //ensure correct info
127  assert( modelConfig [ "backend" ] == "tensorflow" );
128

132 
133  bool isFirst = true;
134  bool isFirstKnown = true;
135  //for each layer, find attributes
136  TensorP<ValueType> input = nullptr;

138  std::string layerType = layerJson [ "class_name" ];
139  LayerP<ValueType, WeightType, DataTensorType, WeightTensorType> layer;
140 
141  std::cout << layerType << std::endl;
142  if( isFirst ){
143  input = grabInputTensor( layerJson [ "config" ], layerType );
144  isFirst = false;
145  }
146 
147  //find correct layer usage
148  if ( layerType == "Conv2D" )
149  layer = grabConv2D( layerJson [ "config" ] );
150  else if ( layerType == "Flatten" )
151  layer = grabFlatten( layerJson [ "config"] );
152  else if ( layerType == "Dense" )
153  layer = grabDense( layerJson [ "config" ] );
154  else if ( layerType == "SimpleRNN" )
155  layer = grabRNN( layerJson [ "config" ] );
156  else if ( layerType == "AveragePooling2D" )
157  layer = grabAveragePooling( layerJson [ "config" ] );
158  else if ( layerType == "ZeroPadding2D" )
159  layer = grabZeroPadding2D( layerJson [ "config" ] );
160  else if( mSkipUnknown )
161  continue;
162  else
163  throw InvalidLayer( layerType );
164 
165  if( isFirstKnown ){
166  layer->input( input );
167  isFirstKnown = false;
168  layer->output( mDtf->create( layer->outputShape() ) );
169  }
170 
171  //add layer
172  model.addLayer( layer );
173 
174  }
175

262  TensorP<ValueType> grabInputTensor( json config, std::string layerType = "" ) {
263  //grab input shape and 'clean' since shapes in keras can be null, we convert them to 1 to avoid issues
264  std::vector<size_t> shapeVector;
265 
266  // embedding layers need some special handling becasue we don't do the embedding. We want to get the outputshape

268  if( layerType == "Embedding" ){
269  // (batch_size, sequence_length, output_dim)
270  shapeVector.push_back( 1 ); // batch_size TODO make it configureable
271  shapeVector.push_back( config[ "input_length" ] );
272  shapeVector.push_back( config[ "output_dim" ] );
273 
274  }else{
275 
276  auto shape = config [ "batch_input_shape" ];
277  for ( auto& el : shape ){
278  if ( el.is_null() )
279  el = 1;
280  shapeVector.push_back( el );
281  }
282  }
283  Shape inputShape( shapeVector );
284  TensorP<ValueType> input = mDtf->create( inputShape );
