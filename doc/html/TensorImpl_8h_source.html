483  size_t toIndex( const std::initializer_list<size_t> & idxs ) {
484  // every index is multiplied with the product of the following dimensions
485  // considering a tensor with shape( x,y,z ) the index i would be calculated as
486  // ( i1 * ( y * z )) + ( i2 * z ) + i3
487  size_t idx = 0;
488  size_t i = 0;
489  for ( auto x : idxs )
490  idx += x * strides [ i++ ];
491  return idx;
